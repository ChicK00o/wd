51 static void get_home( void )
52 {
53  int success = 0;
54 
55 #if defined _WIN32
56  char homedir[MAX_PATH];
57  success = SUCCEEDED(SHGetFolderPath(NULL, CSIDL_PROFILE, NULL, 0, homedir));
58 #else
59  /* Try and retrieve the home from the environment first */
60  char *homedir = getenv("HOME");
61 
62  if( homedir == NULL ) {
63  /* Couldn't get user's home directory from the environment so
64  try the user database */
65  uid_t uid = getuid();
66  struct passwd *pw = getpwuid(uid);
67  if (pw != NULL) {
68  homedir = pw->pw_dir;
69  }
70  }
71  if( homedir != NULL ) {
72  success = 1;
73  }
74 #endif
75 
76  if( success ) {
77  /* Stitch the home directory and default filename together into list_fn
78  */
79 
80  size_t home_size = strlen( homedir );
81  size_t complete = home_size +
82  strlen( DEFAULT_LIST_FILE ) +
83  1U;
84  list_fn = realloc( list_fn, complete );
85  strcpy( list_fn, homedir );
86  strcpy( &(list_fn[ home_size ]), DEFAULT_LIST_FILE);
87  list_fn[ complete -1 ] = 0;
88  } else {
89  /* TODO: Deal with not having a home directory */
90  }
91 }

126 int process_cmdln( const int argc, char* const argv[] ) {
127  int arg_loop;
128  int ret_val = 1;
129 
130  /* Start loop at 1 - index 0 is name of executable */
131  for( arg_loop = 1; arg_loop < argc; arg_loop++ )
132  {
133  char* this_arg = argv[ arg_loop ];
134  if( 0 == strcmp( this_arg, "-v" ) ) {
135  fprintf( stdout, "%s\n", VERSION_STRING );
136  } else if( 0 == strcmp( this_arg, "-h" ) ) {
137  show_help( argv[0] );
138  } else if( 0 == strcmp( this_arg, "-p" ) ) {
139  wd_prompt = 1;
140  } else if( 0 == strcmp( this_arg, "-s" ) ) {
141  if(( arg_loop + 1 ) < argc ) {
142  arg_loop++;
143  /* TODO: Validate length of argument? */
144  switch( argv[ arg_loop ][0] ) {
145  case 'c':
146  wd_dir_form = WD_DIRFORM_CYGWIN;
147  break;
148  default:
149  fprintf( stdout, "%s: %s\n", UNRECOGNISED_PARAM_STRING, this_arg );
150  ret_val = 0;
151  break;
152  }
153  } else {
154  fprintf( stdout, "%s: %s\n", NEED_PARAMETER_STRING, this_arg );
155  ret_val = 0;
156  }
157  } else if( 0 == strcmp( this_arg, "-d" ) ) {
158  wd_oper = WD_OPER_DUMP;
159  } else if( 0 == strcmp( this_arg, "-l" ) ) {
160  wd_oper = WD_OPER_LIST;
161  } else if( 0 == strcmp( this_arg, "-n" ) ) {
162  if((( arg_loop + 1 ) < argc ) &&
163  ( argv[ arg_loop + 1 ][0] != '-' )) {
164  arg_loop++;
165  wd_oper = WD_OPER_GET_BY_BM_NAME;
166  wd_bookmark_name = argv[ arg_loop ];
167  } else {
168  fprintf( stdout, "%s: %s\n", NEED_PARAMETER_STRING, this_arg );
169  ret_val = 0;
170  }
171  } else if(( 0 == strcmp( this_arg, "-a" )) ||
172  ( 0 == strcmp( this_arg, "-r" )) ) {
173  if( wd_oper == WD_OPER_NONE ) {
174  switch( this_arg[ 1 ] ) {
175  case 'a':
176  wd_oper = WD_OPER_ADD;
177  break;
178  case 'r':
179  wd_oper = WD_OPER_REMOVE;
180  break;
181  }
182 
183  /* Check to see if there's an argument to this command */
184  if((( arg_loop + 1 ) < argc ) &&
185  ( argv[ arg_loop + 1 ][0] != '-' )) {
186  arg_loop++;
187 
188  /* TODO: Add a switch to prevent the path being made
189  absolute? */
190 #if defined _WIN32
191  GetFullPathName( argv[ arg_loop ],
192  MAXPATHLEN,
193  wd_oper_dir,
194  NULL );
195 #else
196  realpath( argv[ arg_loop ], wd_oper_dir );
197 #endif
198 
199  /* TODO: This only really makes sense for an add operation
200  */
201  if((( arg_loop + 1 ) < argc ) &&
202  ( argv[ arg_loop + 1 ][0] != '-' )) {
203  arg_loop++;
204  wd_bookmark_name = argv[ arg_loop ];
205  }
206  } else {
207  /* No argument .. use the CWD */
208  getcwd( wd_oper_dir, MAXPATHLEN );
209  }
210  } else {
211  fprintf( stdout, "%s: %s\n", INCOMPATIBLE_OP_STRING, this_arg );
212 
213  /* Check to see if there's an argument to this command */
214  if((( arg_loop + 1 ) < argc ) &&
215  ( argv[ arg_loop + 1 ][0] != '-' )) {
216  arg_loop++;
217  }
218  }
219  } else if( 0 == strcmp( this_arg, "-f" ) ) {
220  arg_loop++;
221  if( arg_loop < argc ) {
222  /* To be consistent, list_fn always points to malloc'd memory
223  rather than just pointing it to the parameter string */
224  list_fn = realloc( list_fn, strlen( argv[ arg_loop ] ));
225  strcpy( list_fn, argv[ arg_loop ] );
226  } else {
227  fprintf( stdout, "%s: %s\n", NEED_PARAMETER_STRING, this_arg );
228  ret_val = 0;
229  }
230  } else {
231  fprintf( stdout, "%s:\n %s\n", UNRECOGNISED_ARG_STRING, this_arg );
232  ret_val = 0;
233  }
234  }
235 
236  return ret_val;
237 }
